271  inline std::string OptionsParser::explodeBundledFlags(const std::string& args) {
272  bool _ = true;
273  std::string parsable(args);
274  std::size_t bundleStart = 0;
275  std::string bundleSequence(" -");
276 
277  //&!off
278  for (;_;) {
279  //&!on
280  bundleStart = parsable.find(bundleSequence, bundleStart);
281  if (bundleStart == -1) break;
282  if (parsable[bundleStart + 2] == '-' && ++bundleStart) continue;
283 
284  std::size_t bundleEnd = parsable.find(' ', bundleStart + 1);
285  std::size_t bundleSize = bundleEnd - bundleStart - 1;
286  if (bundleSize <= 1 && ++bundleStart) continue;
287 
288  std::string bundle = parsable.substr(bundleStart, bundleSize + 1);
289  parsable.erase(bundleStart, bundleSize + 1);
290  std::stringstream explodedBundleStream;
291  for (const auto& ch : bundle) {
292  unless (ch == ' ' || ch == '-') {
293  explodedBundleStream << " -" << ch << ' ';
294  }
295  }
296  std::string explodedBundle = explodedBundleStream.str();
297 
298  parsable.insert(bundleStart, explodedBundle);
299  bundleStart++;
300  }
301 
302  return parsable;
303  }
