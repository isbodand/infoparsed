127  inline std::string OptionsParser::parse(const std::string& args) {
128  std::string parsable(equalizeWhitespace(explodeBundledFlags(args)));
129  for (const auto& handler : optionHandlers) {
130  parsable = handler.second.second(handler.second.first, parsable);
131  }
132  return parsable;
133  }

139  inline std::string OptionsParser::explodeBundledFlags(const std::string& args) {
140  bool _ = true;
141  std::string parsable(args);
142  std::size_t bundleStart = 0;
143  std::string bundleSequence(" -");
144 
145  //&!off
146  for (;_;) {
147  //&!on
148  bundleStart = parsable.find(bundleSequence, bundleStart);
149  if (bundleStart == -1) break;
150  if (parsable[bundleStart + 2] == '-' && ++bundleStart) continue;
151 
152  std::size_t bundleEnd = parsable.find(' ', bundleStart + 1);
153  std::size_t bundleSize = bundleEnd - bundleStart - 1;
154  if (bundleSize <= 1 && ++bundleStart) continue;
155 
156  std::string bundle = parsable.substr(bundleStart, bundleSize + 1);
157  parsable.erase(bundleStart, bundleSize + 1);
158  std::stringstream explodedBundleStream;
159  for (const auto& ch : bundle) {
160  unless (ch == ' ' || ch == '-') {
161  explodedBundleStream << " -" << ch << ' ';
162  }
163  }
164  std::string explodedBundle = explodedBundleStream.str();
165 
166  parsable.insert(bundleStart, explodedBundle);
167  bundleStart++;
168  }
169 
170  return parsable;
171  }
