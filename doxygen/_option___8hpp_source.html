73  /**
74  * Constructs the Option_ with the provided
75  * names and function callback.
76  *
77  * Let the input value be `f` of type `F*`.
78  * `F` shall be either 1) `R(P)`; or 2) `R(P, Ps)`.<br />
79  * In case of `R(P)`:
80  * - `P` shall be either 1) `void`; or 2) std or c-style string; or
81  * 3) any type for which `can_stream<P>` is `true`.
82  * -# If `P` is `void`, `f` is called like `(*f)()`.
83  * -# If `P` is a string (std, or c-string) `f` will be called directly
84  * with the parsed value. The empty string is a viable value, for example
85  * in case of parsing `"--option="`.
86  * -# Otherwise according to the parsing rules used by the declared
87  * `operator>>(std::istream&, P&)`, let `val` be the value of
88  * type `P` containing the value extraced from said operator;
89  * then `f` will be called as `(*f)(val)`.
90  * - `R` shall be either 1) a `void` type; or 2) pointer type of `pR*`; or
91  * 3) any type which can be cast to `int`; or 4) any type which can be converted
92  * to `bool`.
93  * -# If the `R` is a possibly cv-qualified `void`, the function is called;
94  * no diagnostic is performed and the function is believed to have
95  * succeeded.
96  * -# If `R` is a pointer type of `pR*`, then the returned value is
97  * checked for `nullptr`. If it is the function is considered to have
98  * failed.
99  * -# If `R` is convertible to `int`, the returned value will be
100  * checked for the return value as in a shell.
101  * Let `val` be the value to be passed to `f` and of type `P` as
102  * described above.
103  * If the expression `((int) (*f)(val)) == 0` is `false`, the function
104  * is considered to have failed.
105  * -# If `R` is convertible to `bool`;
106  * Let `val` be the value to be passed to `f` and of type `P` as
107  * described above.
108  * If the expression ((bool) (*f)(val)) is `false`, the function is
109  * considered to have failed.
110  * -# Otherwise no diagnostic is performed and the function is believed
111  * to have succeeded.
112  *
113  * In case of `R(P, Ps)`:<br />
114  * `P` & `R` behave the same as it is described above for `R(P)`.
115  * `Ps` is std or c-style string and the direct parsed value
116  * is supplied there in the applicable type. If `Ps` is of type `pPs*`,
117  * then the input value will be `((pPs*) val.c_str())`. Otherwise the library
118  * tries passing in a default constructed object with the value of `Ps{}`.
119  *
120  * In case the function is deemed to have failed, it is run once
121  * more, without any diagnostic. If it were to fail a second time,
122  * it shall stay as failed, the library won't care.
123  *
124  * @param[in] names The names of the param split by '|'
125  * @param[out] func The function callback to call with the found value
126  */
127  Option_(OptionString names, const std::function<R(Args...)>& func);

283  template<class T, class R, class... Args>
284  int Option_<T, R, Args...>::handleFlagParse(std::string& parsee,
285  StrCIter f,
286  StrCIter l) const {
287  auto lp = std::distance(parsee.begin(), l);
288  auto fp = std::distance(parsee.begin(), f);
289  int bonus = fp + 2 != lp;
290  auto evalVal = [](const std::string& val) {
291  if (val == "yes" || val == "true") { // true values
292  return true;
293  } else if (val == "no" || val == "false") { // false values
294  return false;
295  } else if (anyOf(val[0], "0123456789")) {
296  //starts with digit means we check numeric truthyness
297  return (bool) std::atoi(val.c_str());
298  } else {
299  //if value is only space it is falsy, otherwise truthy
300  return !std::all_of(val.begin(), val.end(), ::isspace);
301  }
302  };
303 
304  // before anyone asks why the fuck did I even consider and then implement
305  // the ability to supply truthyness evaluation as a value to flags;
306  // it exists to make this possible:
307  // ./prog --use-wget=$(which wget 2>/dev/null || echo -n)
308  switch (*l) {
309  case ' ': // --flagtext is not valid for --flag
310  parsee.erase(f - bonus, l + 1);
311  callCallback("1");
312  return 1;
313  case '=': {
314  // +1 for we need not the =
315  auto val = parsee.substr(lp + 1, parsee.find(' ', lp) - (lp + 1));
316  to_lower(val);
317  callCallback(std::to_string((int) evalVal(val)));
318  parsee.erase(fp - bonus, lp - (fp - bonus) + 2 + val.size()); // +2 for '=' & trailing space
319  return 1;
320  }
321  case ':': {
322  if (l + 1 == parsee.end()) {
323  callCallback("0");
324  parsee.erase(fp - bonus, lp - (fp - bonus) + 1); // +1 for ':'
325  return 1;
326  }
327  auto firstNonSpace = parsee.find_first_not_of(' ', lp + 1); // +1 for we need not the :
328  auto whitespaces = firstNonSpace - (lp + 1);
329  auto endOfValue = parsee.find(' ', firstNonSpace);
330  auto val = parsee.substr(firstNonSpace, endOfValue - firstNonSpace);
331  to_lower(val);
332  callCallback(std::to_string((int) evalVal(val)));
333  parsee.erase(fp - bonus,
334  lp - (fp - bonus) + whitespaces + 2 + val.size()); // +2 for ':' & trailing space
335  return 1;
336  }
337  default:
338  if (l == parsee.end()) {
339  parsee.erase(f - bonus, l);
340  callCallback("1");
341  return 1;
342  }
343  return 0;
344  }
345  }

347  template<class T, class R, class... Args>
348  int Option_<T, R, Args...>::handleOptionalNegatedFlagParse(std::string& parsee,
349  StrCIter f,
350  StrCIter l) const {
351  auto lp = std::distance(parsee.begin(), l);
352  auto fp = std::distance(parsee.begin(), f);
353  std::string noStr = "--no";
354  bool match = false;
355  for (int i = 0; i < noStr.size(); ++i) {
356  match = *(f - (i + 1)) == noStr[noStr.size() - (i + 1)];
357  }
358  if (match) { // negated flag invocation
359  callCallback("0");
360  parsee.erase(fp - noStr.size(),
361  lp - fp + 1 + noStr.size()); //+1 for trailing space
362  return 1;
363  }
364  // no negation just random 'o' before flag
365  // note that *o-flag is invalid
366  // and short _options are not negatable
367  /* so just */ return 0;
368  }

370  template<class T, class R, class... Args>
371  int Option_<T, R, Args...>::parseFlag(std::string& parsee,
372  std::pair<StrCIter, StrCIter> match) const {
373  auto& f = match.first;
374  auto& l = match.second;
375  if (f == l) { // 404
376  return 0;
377  }
378 
379  if (*(f - 1) == '-') { // long flag invocation
380  return handleFlagParse(parsee, f, l);
381  } else if (*(f - 1) == 'o') { // negated flag invocation?
382  return handleOptionalNegatedFlagParse(parsee, f, l);
383  } else if (f + 2 == l) { // short flag
384  return handleFlagParse(parsee, f, l);
385  }
386  return 0;
387  }

389  template<class T, class R, class... Args>
390  int Option_<T, R, Args...>::parseValue(std::string& parsee,
391  std::pair<StrCIter, StrCIter> match) const {
392  auto& f = match.first;
393  auto& l = match.second;
394  if (f == l) { // 404
395  return 0;
396  }
397 
398  auto lp = std::distance(parsee.begin(), l);
399  auto fp = std::distance(parsee.begin(), f);
400  int bonus = fp + 2 != lp;
401  int addendum = 1; // to skip ':' or leading ' '
402 
403  switch (*l) {
404  case '=': {
405  // +1 for we need not the =
406  auto val = parsee.substr(lp + 1, parsee.find(' ', lp) - (lp + 1));
407  callCallback(val);
408  parsee.erase(fp - bonus, lp - (fp - bonus) + 2 + val.size()); // +2 for '=' & trailing space
409  return 1;
410  }
411  default:
412  if (l == parsee.end()) {
413  parsee.erase(f - bonus, l);
414  callCallback("");
415  return 1;
416  }
417  addendum--;
418  [[fallthrough]];
419  case ' ': [[fallthrough]];
420  case ':': {
421  if (l + 1 == parsee.end()) {
422  callCallback("");
423  parsee.erase(fp - bonus, lp - (fp - bonus) + addendum);
424  return 1;
425  }
426  auto firstNonSpace = parsee.find_first_not_of(' ', lp + addendum);
427  auto whitespaces = firstNonSpace - (lp + addendum);
428  auto endOfValue = parsee.find(' ', firstNonSpace);
429  auto val = parsee.substr(firstNonSpace, endOfValue - firstNonSpace);
430  callCallback(val);
431  parsee.erase(fp - bonus,
432  lp - (fp - bonus) + whitespaces + addendum + 1 + val.size()); // +1 for trailing space
433  return 1;
434  }
435  }

438  template<class T, class R, class... Args>
439  std::string Option_<T, R, Args...>::iterateNamesOnWith(std::string parsee,
440  bool flag) const {
441  typedef std::string::size_type strsize_t;
442  auto useBMH = []([[maybe_unused]] strsize_t parsedSize,
443  [[maybe_unused]] strsize_t parseeSize) {
444  return parsedSize > 10;
445  };
446 
447  auto getParser = [&](bool flag_) {
448  if (flag_) {
449  return &Option_::parseFlag;
450  } else {
451  return &Option_::parseValue;
452  }
453  };
454 
455  for (auto &&[name, kmp, bmh] : *names) {
456  if (useBMH(name.size(), parsee.size())) {
457  // If relative long jumps are possible use boyer-moore-horspool
458  (this->*getParser(flag))(parsee,
459  bmh(name)(parsee.begin(), parsee.end()));
460  } else {
461  // otherwise fall-back on knuth-morris-pratt
462  (this->*getParser(flag))(parsee,
463  kmp(name)(parsee.begin(), parsee.end()));
464  }
465  }
466  return parsee;
467  }

496  template<class T, class R, class... Args>
497  void Option_<T, R, Args...>::callCallback(const std::string& value) const {
498  if constexpr (std::is_same_v<T, none>) {
499  auto checkReturnAndRetryIfNeed = [](const std::function<R(Args...)>& f,
500  Args... args) {
501 // if (config::RetryFailedCallback) {
502  if constexpr (std::is_pointer_v<R>) {
503  if (config::DeleteCallbackReturn) {
504  auto ptr = f(args...);
505  if (ptr == nullptr) {
506  ptr = f(args...); // Retry failed func
507  }
508  delete ptr;
509  } else {
510  if (f(args...) == nullptr) {
511  f(args...); // Retry failed func
512  }
513  }
514  } else if constexpr (std::is_convertible_v<R, int>) {
515  unless (((int) f(args...)) == 0) {
516  f(args...); // Retry failed func
517  }
518  } else if constexpr (std::is_convertible_v<R, bool>) {
519  unless ((bool) f(args...)) {
520  f(args...); // Retry failed func
521  }
522  } else {
523  f(args...);
524  }
525 // } else f();
526  };
527 
528  using Arg1 = std::remove_cv_t<std::remove_reference_t<typename fP<Args...>::Type>>;
529  using Arg2 = std::remove_cv_t<std::remove_reference_t<typename sP<Args...>::Type>>;
530 
531  auto makeArg = [](const std::string& value) -> Arg1 {
532  if constexpr (std::is_same_v<Arg1, std::string>) {
533  // String is output directly
534  return value;
535  }
536  std::istringstream ss(value);
537  Arg1 arg1;
538  ss >> arg1;
539  return arg1;
540  };
541 
542  // Give me switch constexpr pls
543  constexpr std::size_t args = sizeof...(Args);
544  if constexpr (args == 0) {
545  checkReturnAndRetryIfNeed(*callback);
546  } else if constexpr (args == 1) {
547  checkReturnAndRetryIfNeed(*callback, makeArg(value));
548  } else if constexpr (args == 2) {
549  if constexpr (std::is_same_v<Arg2, std::string>) {
550  // exporter takes 2 values
551  checkReturnAndRetryIfNeed(*callback, makeArg(value), value);
552  } else if constexpr (std::is_pointer_v<Arg2>) {
553  // You asked for it
554  checkReturnAndRetryIfNeed(*callback, makeArg(value), (Arg2) value.c_str());
555  } else // Hope this makes sense
556  checkReturnAndRetryIfNeed(*callback, makeArg(value), Arg2{});
557  } else if (!config::FailSilently) {
558  throw bad_function_callback(sizeof...(Args));
559  }
560  } else {
561  auto td = mkTypeD(exporter);
562  using TDType = decltype(td);
563  using Typ =
564  typename std::remove_pointer_t<std::tuple_element_t<0, TDType>>;
565  using SecTyp = std::tuple_element_t<1, TDType>;
566 
567  if constexpr (std::is_same_v<SecTyp, bool>) {
568  using Re = typename Typ::Ret;
569  using Arg1 = std::remove_cv_t<std::remove_reference_t<typename Typ::Arg0>>;
570  using Arg2 = std::remove_cv_t<std::remove_reference_t<typename Typ::Arg1>>;
571 
572  auto callF = [&]() -> Re {
573  auto makeArg = [](const std::string& value) -> Arg1 {
574  if constexpr (std::is_same_v<std::remove_cv_t<std::remove_reference_t<Arg1>>,
575  std::string>) {
576  // String is output directly
577  return value;
578  }
579  std::istringstream ss(value);
580  Arg1 arg1;
581  ss >> arg1;
582  return arg1;
583  };
584 
585  if constexpr (std::is_same_v<Arg1, none>) {
586  // exporter takes no parameters
587  return (*exporter)();
588  } else if constexpr (std::is_same_v<Arg2, none>) {
589  // exporter takes 1 parameter
590  return (*exporter)(makeArg(value));
591  } else if constexpr (std::is_same_v<std::remove_reference_t<std::remove_cv_t<Arg2>>,
592  std::string>) {
593  // exporter takes 2 values
594  return (*exporter)(makeArg(value), value);
595  } else if constexpr (std::is_pointer_v<Arg2>) {
596  // You asked for it
597  return (*exporter)(makeArg(value), (Arg2) value.c_str());
598  } else
599  // Hope this makes sense
600  return (*exporter)(makeArg(value), Arg2{});
601  };
602 
603  if constexpr (std::is_pointer_v<Re>) {
604  if (callF() == nullptr) {
605  callF(); // Retry failed func
606  }
607  }
608  if constexpr (std::is_convertible_v<Re, int>) {
609  unless (((int) callF()) == 0) {
610  callF(); // Retry failed func
611  }
612  }
613  if constexpr (std::is_convertible_v<Re, bool>) {
614  unless ((bool) callF()) {
615  callF(); // Retry failed func
616  }
617  }
618  // Not checking success
619  callF();
620  } else {
621  if constexpr (std::is_same_v<std::remove_reference_t<std::remove_cv_t<T>>,
622  std::string>) {
623  // String is output directly
624  *exporter = value;
625  } else if (value != "") {
626  std::istringstream ss(value);
627  ss >> *exporter;
628  } else {
629  T val{};
630  *exporter = val;
631  }
632  }
633 
634  delete std::get<0>(td);
635  }
636  }
